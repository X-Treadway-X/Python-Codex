Example 6: Basic Class and Object

    def __init__(self, name):
        self.name = name

        print(f"{self.name} says Woof!")

Example 7: Handling Exceptions

try:
    x = int(input("Enter a number: "))
    print(10 / x)
except ZeroDivisionError:
    print("Cannot divide by zero.")
except ValueError:
    print("Invalid input.")
